if else pipes

A nice way to use if else within pipes is (see here):

library(magrittr)
runif(1) %>% magrittr::is_greater_than(0.5) %>% {if(.) "foo" else "bar"}
## [1] "bar"

Alternatively, the when function from the purrr package can be used:

library(purrr, quietly = TRUE, warn.conflicts = FALSE)
runif(1) %>% purrr::when(. <= 0.5 ~ "foo", 
                         . >  0.5 ~ "bar") 
## [1] "foo"

A nice way to use a vector of inputs is provided by the case_when function from dplyr together with mutate or transmute:

library(dplyr,  quietly = TRUE, warn.conflicts = FALSE)
library(tibble, quietly = TRUE, warn.conflicts = FALSE)
runif(100) %>% 
  magrittr::is_greater_than(0.5) %>% 
  tibble::tibble(a = .) %>% 
  dplyr::transmute(., foobar = case_when(a <= 0.5 ~ "fu", a >  0.5 ~ "bar"))
## # A tibble: 100 x 1

##    <chr> 
##  1 bar

##  3 fu    
##  4 fu    
##  5 fu    
##  6 fu
