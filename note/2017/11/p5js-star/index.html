p5js: star

Another little Javascript-animation based on Gaussian ellipses.

Code:

var c_size = 450
var half_c_size = 0.5*c_size;
var c_pad = c_size - 20
var time = 0.0
var increment = 0.002;
var time_to_update = time + 2*increment;
var increment_to_update = 0.03;
var n = 0
var m = 0

function setup() {
  background("#DDDCD3");
  var canvas = createCanvas(c_size, c_size);
  canvas.parent('sketch-holder');

  // info:
  textSize(19);
  stroke(221,220,211);
  fill(221,220,211);
  ellipse(half_c_size, half_c_size, c_pad/4, c_pad/4);
  //
  fill("#EEEDE4");
  text("    ~68%  \n of the radii", c_pad/2-40, c_size/2);
  // decoration:
  // outer circles
    fill(221,220,211,0);
    stroke(221,220,211);
    strokeWeight(3);
    ellipse(half_c_size, half_c_size, c_pad/3, c_pad/3);
    ellipse(half_c_size, half_c_size, c_pad/2, c_pad/2);
    ellipse(half_c_size, half_c_size, c_pad, c_pad);
    // outer ellipses
    strokeWeight(1);
    ellipse(half_c_size, half_c_size, c_pad/3, c_pad/4);
    ellipse(half_c_size, half_c_size, c_pad/4, c_pad/3);
    ellipse(half_c_size, half_c_size, c_pad/2, c_pad/3);
    ellipse(half_c_size, half_c_size, c_pad/3, c_pad/2);
    ellipse(half_c_size, half_c_size, c_pad  , c_pad/2);
    ellipse(half_c_size, half_c_size, c_pad/2, c_pad  );
}

function draw() {
translate(width/2, height/2);
rotate(frameCount / 50.0);
  noStroke();
  time += increment;
  if (time >= time_to_update) {
    n = abs(randomGaussian(1,c_pad/4));
    m = abs(randomGaussian(1,c_pad/4));
    time_to_update += increment_to_update

    fill(30, 144, 255, 20);
    ellipse(0, 0, n, m);
  }
}
