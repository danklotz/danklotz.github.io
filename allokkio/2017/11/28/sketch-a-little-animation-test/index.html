Sketch: A little Animation-Test

My first try at p5js. A little animation that plots a circle, with a radius determined by Guassian noise.

Code:

// variable definitions:
var c_size = 450
var c_pad = c_size - 20
var time = 0.0;
var increment = 0.001;
var time_to_update = time + 2*increment;
var increment_to_update = 0.01;
var n = 0

// canvas setup:
function setup() {
  background("#DDDCD3");
  var canvas = createCanvas(c_size, c_size);
  canvas.parent('sketch-holder');
}

// generation and drawing:
function draw() {
  noStroke();
  time += increment;
  if (time >= time_to_update) {
    n = abs(randomGaussian(1,c_pad/2));
    time_to_update += increment_to_update

    fill(30, 144, 255, 10);
    ellipse(c_size/2, c_size/2, n, n);
  }
  stroke("#ffa500")
  strokeWeight(1)
  fill(221,220,211,0);
  ellipse(c_size/2, c_size/2, c_pad/2, c_pad/2);
  ellipse(c_size/2, c_size/2 ,c_pad, c_pad);

  // text
  textSize(19);
  noStroke( );
  fill("#ffa500");
  text("~68% of \n  samples", c_pad/4 + 18, c_size/2);
  text("~95% of \n  samples", 18, c_size/2);
}
