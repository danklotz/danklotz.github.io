Sketch: Doodle Flower

Code:

var c_size = 450
var midpoint = 0.5*c_size;
var c_pad = c_size - 20
var alpha_decay = 20
var easing = 0.03;
var x = midpoint
var y = midpoint

var red = 30
var green = 144
var blue = 255

var update_time = 0
var update_step = 100

function setup() {
  background("#DDDCD3");
  var canvas = createCanvas(c_size, c_size);
  canvas.parent('sketch-holder');

}

function mousePressed() {
  red = 30
  green = 144
  blue = 255
}

function draw() {

  if (frameCount > update_time) {
    update_time += update_step
    red = round( random(1,244) )
    green = round( random(1,244) )
    blue = round( random(1,244) )
  }

  //square
  stroke("#DDDCD3")
  fill("#DDDCD3")
  rect(0,0,c_size-1,c_size-1);
  // outer circles
  fill("#EEEDE4");
  stroke("#EEEDE4")
  ellipse(0,0,c_size,c_size);
  ellipse(0,c_size,c_size,c_size);
  ellipse(c_size,0,c_size,c_size);
  ellipse(c_size,c_size,c_size,c_size);
  // line
  stroke(red, green, blue);
  line(midpoint, midpoint, x, y)
  // cicrles 1
  fill(red, green, blue);
  stroke(red, green, blue);
    ellipse(midpoint,
            midpoint,
            c_pad/6,
            c_pad/6);
    ellipse(x, y, 25, 25);
    ellipse(0.5*(midpoint+x),
            0.5*(midpoint+y),
            c_pad/8,
            c_pad/8);
  fill(221, 220, 211, 0);
  strokeWeight(3);
  stroke(red, green, blue, 255-alpha_decay);
    ellipse(0.65*midpoint + 0.35*x,
            0.65*midpoint + 0.35*y,
            c_pad/4,
            c_pad/4);
  stroke(red, green, blue, 255 - 2*alpha_decay);
    ellipse(0.8*midpoint + 0.2*x,
            0.8*midpoint + 0.2*y,
            c_pad/3,
            c_pad/3);
  stroke(red, green, blue, 255 - 4*alpha_decay);
    ellipse(0.9*midpoint + 0.1*x,
            0.9*midpoint + 0.1*y,
            c_pad/2,
            c_pad/2);
  stroke(red, green, blue, 255 - 6*alpha_decay);
    ellipse(0.95*midpoint + 0.05*x,
            0.95*midpoint + 0.05*y,
            c_pad,
            c_pad);
  // ellipses 1
  strokeWeight(1);
    ellipse(midpoint, midpoint, c_pad/3, c_pad/4);
    ellipse(midpoint, midpoint, c_pad/4, c_pad/3);
    ellipse(midpoint, midpoint, c_pad/2, c_pad/3);
    ellipse(midpoint, midpoint, c_pad/3, c_pad/2);
    ellipse(midpoint, midpoint, c_pad  , c_pad/2);
    ellipse(midpoint, midpoint, c_pad/2, c_pad  );
  // cicles 2
  strokeWeight(2);
  stroke(red, green, blue, 255 - 8*alpha_decay);
    ellipse(0, midpoint, c_size, c_size);
    ellipse(c_size, midpoint, c_size, c_size);
    ellipse(midpoint, 0, c_size, c_size);
    ellipse(midpoint, c_size, c_size, c_size);
  // update position
  x += (mouseX - x) * easing;
  y += (mouseY - y) * easing;

}
